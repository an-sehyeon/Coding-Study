<script>
    // querySelector("선택자") : 선택자가 포함된 첫 번째 태그
    // querySelectorAll("선택자") : 선택자가 포함된 전체
    const li_odd = document.querySelectorAll("li.odd");
    const li_even = document.querySelectorAll("li.even");
    const li_all = document.querySelectorAll("li");
    console.log(typeof(li_odd));

    li_odd.forEach((li, idx) => {
        li.style.color = "red";
        if(idx==2){
            li.style.color = "crimson";
        }
    });

    // list 중 odd는 magenta, 다른 것은 violet으로 출력
    // getAttribute : 속성값을 얻어내는 함수
    // getAttribute("class") == "odd"  : hint
    // 1단계 : 돌아가게만 하시오
    // 2단계 : 한 줄로 작성하시오
    li_all.forEach((li) => {
        if(li.getAttribute("class")=="odd"){
            li.style.color = "magenta";
        }
        else {
            li.style.color = "violet";
        }
    });

    // 한 줄 버전
    // li.style.color = (li.getAttribute("class")=="odd")?"gray":"darkblue"    
    li_all.forEach((li)=>{li.style.color=(li.getAttribute("class")=="odd")?"gray":"darkblue"});
    
    // createDiv : element 추가
    function createDiv(){
        const div = document.createElement("div");  // <div></div>
        div.innerHTML = "생성할테니 닥달하지마요";
        div.style.background = "yellow";
        document.body.appendChild(div);

        div.addEventListener("dblclick", function(){
            this.innerHTML = "야";
            this.parentElement.removeChild(this);
        });
    }

    // <input type="button" value="New Style?">
    const input = document.querySelector("input[value='New Style?']");
    const span = document.getElementsByTagName("span")[0];

    input.addEventListener("click", function(event){
        console.log(event);
        span.style.background = "darkblue";
        span.style.color = "white";
        span.style.fontSize = "32px";
        span.style.border = "3px dotted lightgreen";
        span.style.display = "inline-block";
        span.style.width = "500px";
        span.style.textAlign = "center";
    });

</script>
