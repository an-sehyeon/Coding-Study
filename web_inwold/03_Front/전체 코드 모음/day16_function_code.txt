    // 1부터 10까지 출력하는 함수
    function printOneToTen(){
        for(let i=1; i<11; i++){
            console.log("[printOneToTen] " + i);
        }
    }

    var result = printOneToTen();
    console.log(result);

    // 1부터 10까지의 합을 출력하는 함수
    function printTot(){
        var tot = 0;
        for(let i=1; i<11; i++){
            tot = tot + i;
        }
        console.log("[printTot] " + tot);
    }

    printTot();

    // 숫자 N을 입력받아 1부터 N까지 더해서 출력하는
    // 함수를 작성해보세요.
    function printTotN(N){
        var tot = 0;
        for(let i=1; i<=N; i++){
            tot = tot + i;
        }
        console.log("[printTotN] " + tot);
    }

    printTotN(1286531283);

    // overloading을 지원하지 않는다.
    // 마지막 함수의 내용으로 결정
    function add(num1, num2){
        console.log("num1 = " + num1);
        console.log("num2 = " + num2);
        return num1 + num2;
    }
    document.write("<h1>[add(a, b)] " + add(1, 3) + "</h1>");

    function add(num1, num2, num3){
        console.log("num3은 무엇인가? " + num3);
        if(num3 == undefined){
            num3 = 0;
        }
        return num1 + num2 + num3;
    }
    document.write("<h1>[add(a, b, c)] " + add(1, 3, 5) + "</h1>");

    // 몇 개를 입력할 지 모르는데 다 더하라고 한다.
    function addN(...nums){
        console.log(typeof(nums));
        console.log("사용자 분이 " + nums.length + "개로 호출하였음");
        let tot = 0;
        for(let i=0; i<nums.length; i++){
            tot += nums[i];
        }
        return tot;
    }
    document.write("<h1>[addN()] " + addN(1,1263,176312,111,8)+"</h1>");

    // 첫 번째 인자는 학교 이름, 두 번째 인자부터는 학생들 이름 N개
    function reserveTraining(schoolName, ...names){
        document.write("<h1>" + schoolName + "학생들 환영합니다.</h1>");
        document.write("<h1>이제까지 무사히 나간 학생이 없습니다.</h1>");
        for(let i=0; i<names.length; i++){
            document.write("<h2>" + names[i] + "</h2>");
        }
    }

    reserveTraining("화산고", "제임스박", "샤론김", "개이슨");

    // 번외 테스트 : 여러 개의 리턴을 하고 싶다.

    // 여러 개를 리턴하고 싶으면 배열에 넣어서 한 번에 해
    function returnMulti(a, b, c){
        var array = new Array();
        array.push("a는 " + a);
        array.push("b는 " + b);
        array.push("c는 " + c);
        return array;
    }
    var x = returnMulti("Ohtani", "Aaron", "Hasung");
    console.log("[returnMulti] " + x);
    console.log("typeof(x) " + typeof(x));
