<!doctype html>
<html>
<head>
    <title>Row Style Example</title>
    <style>                                                     /* 테이블의 모든 헤더 셀과 데이터 셀에 일괄적으로 적용 */
        th, td{
            width: 100px;                                       /* 셀 너비 100px 설정 */ 
            height: 15px;                                       /* 셀 높이 15px 설정 */
            padding: 15px;                                      /* 셀 내부 여백 15px 설정 */
            text-align: center;                                 /* 텍스트 중앙 정렬 */
        }
        thead {background: #e3f2fd;}                          /* 테이블 헤더 배경색 설정 */  
        tbody {background: #e0f2f1;}                          /* 테이블 본문 배경색 설정 */  
        tfoot {background: #ffcdd2;}                          /* 테이블 푸터(최하단) 배경색 설정 */  
    </style>
</head>
<body>
    <table border="1">                                          <!-- 테두리가 1인 테이블 생성 -->
        <thead>                                                 <!-- 테이블의 헤더 섹션 -->
            <tr>
                <th>tagname</th>                                <!-- 첫 번째 헤더 셀 -->
                <th>region</th>
                <th>screenread</th>
                <th>misc</th>
            </tr>
        </thead>
        <tbody>                                                 <!-- 테이블의 본문 섹션 -->
            <caption>                                           
                <strong>                                        <!-- 테이블 제목, 강조 -->
                    &lt;thead&gt;&lt;tbody&gt;&lt;thoot&gt;
                </strong>
            </caption>
            <tr>                                                <!-- 첫 번째 행-->
                <td>thead</td>                                  <!-- 첫 번째 행의 첫 번째 셀의 데이터 -->
                <td>제목</td>
                <td rowspan="3">O</td>                          <!-- screenread 열을 3행에 걸쳐 병합--> -->
                <td style="background: yellow;"></td>         <!-- 첫 번째 행의 네 번째 셀, 배경색이 노란색으로 설정된 빈 셀-->
            </tr>
            <tr>                                                <!-- 두 번째 행-->
                <td>tbody</td>                                  <!-- 두 번째 행의 첫 번째 셀 -->
                <td>본문</td>

                <td style="background: crimson;"></td>        <!-- 두 번째 행의 네 번째 셀, 배경색이 진홍색으로 설정된 빈 셀 -->
            </tr>
            <tr>                                                <!-- 세 번째 행-->
                <td>tfoot</td>
                <td>요약</td>

                <td style="background: lightblue;"></td>      <!-- 세 번째 행의 네 번째 셀, 배경색이 연한 파랑색으로 설정된 빈 셀 -->
            </tr>
        </tbody>
        <tfoot>                                                 <!-- 테이블의 푸터 섹션 -->
            <tr>
                <td colspan="4">Summary</td>                    <!-- Summary 텍스트가 4개의 열을 병합하여 표시 -->            
            </tr>
        </tfoot>
    </table>
</body>
</html>
